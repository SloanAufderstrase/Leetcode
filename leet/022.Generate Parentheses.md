Description:

Given n pairs of parentheses, write a function to generate all combinations of well-formed parentheses.

For example, given n = 3, a solution set is:

>[
  "((()))",
  "(()())",
  "(())()",
  "()(())",
  "()()()"
]

Solution:

## 回溯

全部都生成，然后判断是否是合法的()对

class Solution {
    public List<String> generateParenthesis(int n) {
        List<String> res = new ArrayList<String>();
        if (n < 0) return res;
        
        List<Character> list = new ArrayList<Character>();
        helper(res, n, list);
        return res;
    }
    
    private void helper(List<String> res, int n, List<Character> list) {
        if (list.size() == 2 * n) {
            StringBuilder sb = new StringBuilder();
            if (checkValid(list) == false) return ;
            for (char c : list) {
                sb.append(c);
            }
            res.add(new String(sb));
            
            return ;
        }
        
        list.add('(');
        helper(res, n, list);
        list.remove(list.size() - 1);
        list.add(')');
        helper(res, n, list);
        list.remove(list.size() - 1);
    }
    
    public boolean checkValid(List<Character> list) {
        Deque<Character> stack = new ArrayDeque<Character>();
        for (int i = 0; i < list.size(); i++) {
            if (list.get(i) == '(') {
                stack.push('(');
            } else if (list.get(i) == ')'){
                if(!stack.isEmpty() && stack.peek() == '(') {
                    stack.pop();
                } else {
                    return false;
                }
            }
        }
        
        return stack.size() == 0 ? true : false;
    }
}

class Solution {
    public List<String> generateParenthesis(int n) {
        if (n < 1) return null;
        List<String> list = new ArrayList<>();
        
        helper(list, new char[2 * n], 0);
        
        return list;
    }
    
    public static void helper(List<String> list, char[] strArr, int index) {
        if (index == strArr.length) {
            String str = new String(strArr);
            if (checkValid(str)) {
                list.add(str);
            }
            
            return ;
        }
        
        strArr[index] = '(';
        helper(list, strArr, index + 1);
        strArr[index] = ')';
        helper(list, strArr, index + 1);
        
    }
    
    public static boolean checkValid(String s) {
        Deque<Character> stack = new ArrayDeque<>();
        
        for(int i = 0; i <  s.length(); i++){
            char ch = s.charAt(i);
            if (ch == '('){
                stack.push(ch);
            } else if(!stack.isEmpty() && stack.peek() == '(') {
                stack.pop();
            } else {
                return false;
            }
        }
        
        return stack.isEmpty();
    }
}
